from qdrant_client import QdrantClient
from qdrant_client.models import Distance, VectorParams

def init_qdrant_collections():
    """Инициализация коллекций Qdrant"""
    client = QdrantClient(
        url=os.getenv("QDRANT_URL", "http://localhost:6333"),
        api_key=os.getenv("QDRANT_API_KEY")
    )
    
    # Коллекция для вакансий
    if not client.collection_exists("vacancies"):
        client.create_collection(
            collection_name="vacancies",
            vectors_config=VectorParams(size=384, distance=Distance.COSINE)
        )
    
    # Коллекция для резюме
    if not client.collection_exists("resumes"):
        client.create_collection(
            collection_name="resumes",
            vectors_config=VectorParams(size=384, distance=Distance.COSINE)
        )
    
    return client

# 5.3. Интеграция в utils/api_client.py:

class APIClient:
    def __init__(self, base_url: str):
        self.base_url = base_url
        self.qdrant = QdrantClient(
            url=os.getenv("QDRANT_URL"),
            api_key=os.getenv("QDRANT_API_KEY")
        )
    
    def search_similar_vacancies(self, resume_embedding: list, limit: int = 10):
        """Поиск похожих вакансий по эмбеддингу резюме"""
        results = self.qdrant.search(
            collection_name="vacancies",
            query_vector=resume_embedding,
            limit=limit
        )
        return [hit.payload for hit in results]
    
    def upsert_resume(self, resume_id: str, embedding: list, metadata: dict):
        """Сохранить эмбеддинг резюме в Qdrant"""
        self.qdrant.upsert(
            collection_name="resumes",
            points=[{"id": resume_id, "vector": embedding, "payload": metadata}]
        )

# 5.4. Генерация эмбеддингов:

# backend/services/embedding_service.py:
from sentence_transformers import SentenceTransformer

class EmbeddingService:
    def __init__(self, model_name: str = "all-MiniLM-L6-v2"):
        self.model = SentenceTransformer(model_name)
    
    def encode_text(self, text: str) -> list:
        """Преобразовать текст в векторное представление"""
        return self.model.encode(text).tolist()
    
    def encode_batch(self, texts: list[str]) -> list[list]:
        """Преобразовать несколько текстов"""
        return self.model.encode(texts).tolist()

from airflow import DAG
from airflow.operators.python import PythonOperator
from datetime import datetime, timedelta
import requests
import pandas as pd

default_args = {
    'owner': 'career_assistant',
    'start_date': datetime(2024, 1, 1),
    'retries': 3,

}

dag = DAG(
    'job_market_etl_pipeline',
    default_args=default_args,
    schedule_interval='0 6 * * *',  # Ежедневно в 6:00
    catchup=False,
)

def scrape_vacancies(**context):
    """Парсинг вакансий с внешних источников"""
    # ##############
    # AIRFLOW PIPELINE PLACEHOLDER
    # Этот раздел собирает данные о вакансиях:
    # - hh.ru API
    # - Adzuna API  
    # - HeadHunter RSS
    # ##############
    
    vacancies = []
    
    # Пример: hh.ru API
    response = requests.get(
        "https://api.hh.ru/vacancies",
        params={'text': 'python developer', 'per_page': 100}
    )
    if response.status_code == 200:
        data = response.json()
        vacancies.extend(data.get('items', []))
    
    # Сохранить в XCom для следующих задач
    context['task_instance'].xcom_push(key='vacancies', value=vacancies)
    return len(vacancies)

def process_and_embed(**context):
    """Обработка и создание эмбеддингов"""
    vacancies = context['task_instance'].xcom_pull(key='vacancies')
    
    # ##############
    # LANGGRAPH INTEGRATION POINT
    # Здесь можно вызвать LangGraph для:
    # - Нормализации данных
    # - Извлечения навыков
    # - Генерации эмбеддингов
    # ##############
    
    processed = []
    for vac in vacancies:
        processed.append({
            'id': vac.get('id'),
            'title': vac.get('name'),
            'company': vac.get('employer', {}).get('name'),
            'description': vac.get('description', ''),
            'skills': extract_skills(vac.get('description', '')),
            'salary': vac.get('salary', {}),
            'location': vac.get('area', {}).get('name'),
        })
    
    context['task_instance'].xcom_push(key='processed', value=processed)
    return len(processed)

def load_to_qdrant(**context):
    """Загрузка в Qdrant"""
    processed = context['task_instance'].xcom_pull(key='processed')
    
    # ##############
    # QDRANT INTEGRATION
    # Загрузка эмбеддингов в векторную базу
    # ##############
    
    from qdrant_client import QdrantClient
    from backend.services.embedding_service import EmbeddingService
    
    client = QdrantClient(url=os.getenv("QDRANT_URL"))
    embedder = EmbeddingService()
    
    for item in processed:
        embedding = embedder.encode_text(f"{item['title']} {item['description']}")
        client.upsert(
            collection_name="vacancies",
            points=[{
                "id": str(item['id']),
                "vector": embedding,
                "payload": item
            }]
        )
    
    return "Loaded to Qdrant"

def trigger_airflow_dag(dag_id: str, conf: dict = None):
    """Запустить DAG в Airflow"""
    response = requests.post(
        f"{os.getenv('AIRFLOW_URL')}/api/v1/dags/{dag_id}/dagRuns",

from langgraph.graph import StateGraph, END

class ResumeState(TypedDict):

    experience: List[dict]

def create_resume_graph():
    """Создать граф анализа резюме"""
    workflow = StateGraph(ResumeState)
    
    # Nodes
    workflow.add_node("parse", parse_resume_node)
    workflow.add_node("extract_skills", extract_skills_node)
    workflow.add_node("analyze_experience", analyze_experience_node)
    workflow.add_node("generate_embedding", generate_embedding_node)
    workflow.add_node("match_vacancies", match_vacancies_node)
    workflow.add_node("generate_recommendations", generate_recommendations_node)
    
    # Edges
    workflow.set_entry_point("parse")
    workflow.add_edge("parse", "extract_skills")
    workflow.add_edge("extract_skills", "analyze_experience")
    workflow.add_edge("analyze_experience", "generate_embedding")
    workflow.add_edge("generate_embedding", "match_vacancies")
    workflow.add_edge("match_vacancies", "generate_recommendations")
    workflow.add_edge("generate_recommendations", END)
    
    return workflow.compile()

def create_interview_graph():
    """Создать граф симуляции интервью"""
    workflow = StateGraph(InterviewState)
    
    workflow.add_node("generate_question", generate_question_node)
    workflow.add_node("evaluate_answer", evaluate_answer_node)
    workflow.add_node("adjust_difficulty", adjust_difficulty_node)
    workflow.add_node("provide_feedback", provide_feedback_node)
    
    workflow.set_entry_point("generate_question")
    workflow.add_edge("generate_question", "evaluate_answer")
    workflow.add_edge("evaluate_answer", "adjust_difficulty")
    workflow.add_edge("adjust_difficulty", "provide_feedback")
    workflow.add_conditional_edges(
        "provide_feedback",
        lambda s: "continue" if s['difficulty'] < 10 else "end",
        {"continue": "generate_question", "end": END}
    )
    
    return workflow.compile()

import logging
import sys

def setup_logger(name: str, level: str = "INFO"):
    logger = logging.getLogger(name)
    logger.setLevel(getattr(logging, level.upper()))
    
    handler = logging.StreamHandler(sys.stdout)
    formatter = logging.Formatter(
        '%(asctime)s - %(name)s - %(levelname)s - %(message)s'
    )
    handler.setFormatter(formatter)
    logger.addHandler(handler)
    
    return logger

# backend/metrics.py:
from prometheus_client import Counter, Histogram, start_http_server

REQUEST_COUNT = Counter('api_requests_total', 'Total API requests', ['endpoint'])
REQUEST_LATENCY = Histogram('api_request_duration_seconds', 'Request latency')

def track_request(endpoint: str):
    def decorator(func):
        def wrapper(*args, **kwargs):
            REQUEST_COUNT.labels(endpoint=endpoint).inc()
            with REQUEST_LATENCY.time():
                return func(*args, **kwargs)
        return wrapper
    return decorator
